## 请求封装

### Alova 配置

**文件**: `frontend/src/service/request.ts`

```typescript
import { createAlova } from 'alova'
import GlobalFetch from 'alova/GlobalFetch'
import VueHook from 'alova/vue'
import { useAuthStore } from '@/store'
import { router } from '@/router'

// 创建 Alova 实例
export const request = createAlova({
  baseURL: '/api',
  statesHook: VueHook,
  requestAdapter: GlobalFetch(),
  
  // 请求拦截器
  beforeRequest(method) {
    // 添加 Token
    const authStore = useAuthStore()
    if (authStore.token) {
      method.config.headers.Authorization = `Bearer ${authStore.token}`
    }
    
    // 添加 Content-Type
    if (!method.config.headers['Content-Type']) {
      method.config.headers['Content-Type'] = 'application/json'
    }
  },
  
  // 响应拦截器
  responded: {
    // 成功响应
    onSuccess: async (response, method) => {
      const data = await response.json()
      
      // 统一错误处理
      if (data.code !== 200) {
        // 处理特定错误码
        if (data.code === 401) {
          // Token 过期，跳转登录
          const authStore = useAuthStore()
          authStore.logout()
          router.push('/login')
          throw new Error(data.message || '登录已过期')
        }
        
        // 显示错误信息
        window.$message?.error(data.message || '请求失败')
        throw new Error(data.message)
      }
      
      return data
    },
    
    // 错误响应
    onError: async (error, method) => {
      window.$message?.error('网络请求失败')
      throw error
    },
    
    // 请求完成（无论成功失败）
    onComplete: async (method) => {
      // 可以在这里关闭 loading
    }
  }
})
```

## 使用示例

### 在组件中使用

```vue
<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { fetchGetUserProfile, fetchUpdateUserProfile } from '@/service'

const userInfo = ref<Entity.User | null>(null)
const loading = ref(false)

// 获取用户信息
async function loadUserProfile() {
  try {
    loading.value = true
    const res = await fetchGetUserProfile()
    userInfo.value = res.data
  } catch (error) {
    console.error('Failed to load profile:', error)
  } finally {
    loading.value = false
  }
}

// 更新用户信息
async function handleUpdate() {
  if (!userInfo.value) return
  
  try {
    await fetchUpdateUserProfile({
      nickname: userInfo.value.nickname,
      email: userInfo.value.email
    })
    window.$message?.success('更新成功')
  } catch (error) {
    console.error('Failed to update:', error)
  }
}

onMounted(() => {
  loadUserProfile()
})
</script>
```

### 在 Store 中使用

```typescript
// src/store/auth.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { fetchLogin, fetchGetUserRoutes } from '@/service'

export const useAuthStore = defineStore('auth', () => {
  // State
  const token = ref<string>('')
  const refreshToken = ref<string>('')
  const userInfo = ref<Entity.User | null>(null)
  const userRoutes = ref<AppRoute.Route[]>([])
  
  // Getters
  const isLoggedIn = computed(() => !!token.value)
  const isAdmin = computed(() => userInfo.value?.role === 'admin')
  
  // Actions
  async function login(credentials: { username: string; password: string }) {
    const res = await fetchLogin(credentials)
    
    token.value = res.data.accessToken
    refreshToken.value = res.data.refreshToken
    userInfo.value = {
      id: res.data.id,
      username: res.data.userName,
      email: res.data.email,
      role: res.data.role[0]
    } as Entity.User
    
    // 获取用户路由
    await loadUserRoutes()
    
    return res.data
  }
  
  async function loadUserRoutes() {
    const res = await fetchGetUserRoutes()
    userRoutes.value = res.data
    return res.data
  }
  
  function logout() {
    token.value = ''
    refreshToken.value = ''
    userInfo.value = null
    userRoutes.value = []
  }
  
  return {
    token,
    refreshToken,
    userInfo,
    userRoutes,
    isLoggedIn,
    isAdmin,
    login,
    logout,
    loadUserRoutes
  }
})
```

## 错误处理

### 全局错误处理

```typescript
// src/service/request.ts
export const request = createAlova({
  // ... 其他配置
  
  responded: {
    onSuccess: async (response) => {
      const data = await response.json()
      
      if (data.code !== 200) {
        // 根据错误码处理
        switch (data.code) {
          case 400:
            window.$message?.error(data.message || '请求参数错误')
            break
          case 401:
            window.$message?.error('登录已过期，请重新登录')
            handleTokenExpired()
            break
          case 403:
            window.$message?.error('没有权限执行此操作')
            break
          case 404:
            window.$message?.error('请求的资源不存在')
            break
          case 500:
            window.$message?.error('服务器内部错误')
            break
          default:
            window.$message?.error(data.message || '请求失败')
        }
        
        throw new Error(data.message)
      }
      
      return data
    },
    
    onError: async (error) => {
      if (error.name === 'TypeError' && error.message === 'Failed to fetch') {
        window.$message?.error('网络连接失败，请检查网络')
      } else {
        window.$message?.error('网络请求失败')
      }
      throw error
    }
  }
})

// Token 过期处理
function handleTokenExpired() {
  const authStore = useAuthStore()
  authStore.logout()
  router.push('/login')
}
```

### Token 刷新机制

```typescript
let isRefreshing = false
let refreshSubscribers: Array<(token: string) => void> = []

function subscribeTokenRefresh(callback: (token: string) => void) {
  refreshSubscribers.push(callback)
}

function onTokenRefreshed(newToken: string) {
  refreshSubscribers.forEach(callback => callback(newToken))
  refreshSubscribers = []
}

async function refreshToken() {
  const authStore = useAuthStore()
  
  try {
    const res = await fetchUpdateToken({ refreshToken: authStore.refreshToken })
    authStore.token = res.data.accessToken
    authStore.refreshToken = res.data.refreshToken
    return res.data.accessToken
  } catch (error) {
    authStore.logout()
    router.push('/login')
    throw error
  }
}

export const request = createAlova({
  responded: {
    onSuccess: async (response, method) => {
      const data = await response.json()
      
      if (data.code === 401) {
        // Token 过期，尝试刷新
        if (!isRefreshing) {
          isRefreshing = true
          
          try {
            const newToken = await refreshToken()
            onTokenRefreshed(newToken)
            isRefreshing = false
            
            // 重试原请求
            method.config.headers.Authorization = `Bearer ${newToken}`
            return method.send()
          } catch (error) {
            isRefreshing = false
            throw error
          }
        } else {
          // 等待刷新完成
          return new Promise((resolve) => {
            subscribeTokenRefresh((newToken) => {
              method.config.headers.Authorization = `Bearer ${newToken}`
              resolve(method.send())
            })
          })
        }
      }
      
      return data
    }
  }
})
```
